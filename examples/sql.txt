select 
    b.city_size_2014 as city_size,
	    sum(case when a.ad_status_id=0 then 1 else 0 end ) as activeTotalAd
		from 
		    ods.ad a
			left join
			    (
				     select 
					         city_id,city_size_2014
							     from
								         ods.dim_city_master
										     )b
				on a.city_id=b.city_id
				where 
				    a.created_julian_date <= '20140430'  and
					    a.category_name_en='ershouqiche' 
						group by
						    b.city_size_2014
