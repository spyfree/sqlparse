select
    user_id,
	    service_type_id,
		    sum(total_money) as total_money,
			    count(1) as orders
				from
				    dds.vw_fact_order
					where
					    julian_pay_date between '20140201' and '20140210' and 
						    status_id <> 1
							group by
							    user_id,
								    service_type_id
